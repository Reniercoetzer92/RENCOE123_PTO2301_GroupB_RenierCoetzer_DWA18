// Cards.js
import React, { useEffect, useState } from 'react';
import PreviewShows from './PreviewShows';
import PreviewSeasons from './PreviewSeasons';

function Cards() {
  const [shows, setShows] = useState([]);

  useEffect(() => {
    // Fetch the list of shows from the API
    fetch('https://podcast-api.netlify.app/')
      .then((response) => response.json())
      .then((data) => setShows(data));
  }, []);

  return (
    <div>
      <h1>Podcast Cards</h1>
      {/* Pass the shows data to child components */}
      <PreviewShows shows={shows} />
      <PreviewSeasons shows={shows} />
    </div>
  );
}

export default Cards;
