import React, { useState } from 'react';

function PreviewShows({ shows, onShowClick }) {
  const handleShowClick = (showId) => {
    // Reset the selected season when a different show is clicked
    onShowClick(showId);
    
    // Reset the value of the select element to ""
    const selectElement = document.querySelector('select');
    if (selectElement) {
        selectElement.value = "";
    }
    };
  
  return (
    <div className="PreviewShows">
      <h2>Preview Shows</h2>
      <ul>
        {shows.map((show) => (
          <div key={show.id} onClick={() => handleShowClick(show.id)}>
            <img src={show.image} alt={show.title} />
          </div>
        ))}
      </ul>
    </div>
  );
}

export default PreviewShows;
